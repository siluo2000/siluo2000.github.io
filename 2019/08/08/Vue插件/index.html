预渲染 Prerendering

prerender-spa-plugin

npm i -D prerender-spa-plugin

在webpack.config.js中：

// Base Usage
const path = require('path')
const PrerenderSPAPlugin = require('prerender-spa-plugin')
const Renderer = PrerenderSPAPlugin.PuppeteerRenderer

module.exports = {
  plugins: [
    // ...
    new PrerenderSPAPlugin({
      staticDir: path.join(__dirname, 'dist'), //你要存放的静态资源目录
      routes: ['/', '/faq'], //设置要对哪个文件进行SEO
      //如果没有这个配置不会进行预编译
      renderer: new Renderer({
        inject: {
          foo: 'bar',
        },
        headless: true,
        //在项目的入口中使用document.dispatchEvent(new Event('render-event'))
        //这个是你生命的事件名
        rendererAfterDocumentEvent: 'render-event'
      })
    })
  ]
}
  • 应该要在生产环境下使用，然后输出预渲染的HTML，方便搜索引擎爬虫抓取工具可以直接查看预渲染的内容。
